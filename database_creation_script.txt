CREATE TABLE player(
	player_id INT PRIMARY KEY,
	nickname VARCHAR NOT NULL,
	"password" VARCHAR NOT NULL,
	email VARCHAR NOT NULL UNIQUE
);

CREATE TABLE "game_match"(
	match_id INT PRIMARY KEY,
	date_time TIMESTAMP, 
	winner VARCHAR NOT NULL,
	loser VARCHAR NOT NULL
);
